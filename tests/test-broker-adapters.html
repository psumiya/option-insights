function displayResult(title, success, data) { const container = document.getElementById('test-results'); const section = document.createElement('div'); section.className = 'test-section'; const resultDiv = document.createElement('div'); resultDiv.className = `test-result ${success ? 'pass' : 'fail'}`; resultDiv.textContent = `${success ? '✓' : '✗'} ${title}`; section.appendChild(resultDiv); if (data) { const pre = document.createElement('pre'); pre.textContent = JSON.stringify(data, null, 2); section.appendChild(pre); } container.appendChild(section);
